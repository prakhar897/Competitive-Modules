Modulo Exponentiation:

	long power(long base, long power) {
	    long result = 1;
	    while (power > 0) {
	        if ((power & 1) == 1) {
	            result = result * base % MODULO;
	        }
	        base = base * base % MODULO;
	        power >>>= 1;
	    }
	    return result;
	}

	public long gcd(long a, long b) {
        return (a % b == 0 ? b : gcd(b, a % b));
    }

    public int gcd(int a, int b) {
        return (a % b == 0 ? b : gcd(b, a % b));
