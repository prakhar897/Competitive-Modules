
class Pair
{
    int key;
    int value;

    public Pair(int key,int value){
    this.key = key;
    this.value = value;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (!(o instanceof Pair)) return false;
        Pair key = (Pair) o;
        return key == key.key && value == key.value;
    }

    @Override
    public int hashCode() {
        int result = key;
        result = 31 * result + value;
        return result;
    }

}
